# 2018-09-08 파이썬 주사위 프로그램 과제



- 수업 시간에 다룬 주사위 던지기 모사 프로그램을 작성하시오.

   

  1. 주사위 던지기 모사 프로그램

   

  2. 던져진 결과를 turtle 그래픽으로 보여주기

   

  3. 그래프의 색과 빈도수 폰트의 크기 조절

   

  4. 위의 기능을 함수로 만들고 이를 동전던지기, 가위바위보 모사로 확장 변경해 보시오.





```python
#201811458 2018-09-08 정종현
#파이썬 주사위 프로그램 실습과 과제.

import sys
import random
import turtle
#주사위 숫자를 받아오는 리스트
num = []
#동전 앞 뒤를 받아오는 리스트
coin1 = []
#주사위 빈도를 받아오는 리스트
frequency = []
#동전 앞 뒤를 받아오는 리스트
coin_frequency = []

#가위바위보 승패 모음 리스트
result = []

color = ["red","orange","yellow","green","blue","purple"]

try:
    print("========================================")
    select = int(input("     1. 주사위게임 \n     2. 동전게임 \n     3. 가위바위보 " "\n========================================\n 할 게임을 선택하세요 : "))
    freq = int(input("몇 회 진행 하시겠습니까?"))
except ValueError:
    print("숫자를 입력해주세요. 프로그램을 종료합니다.")
    sys.exit(0)

def dice_play():
    for i in range (freq):
        dice = random.randint(1,6)
        num.append(dice)

    for i in range (6):
        print (i+1,"이(가) 나온 횟수 = ",num.count(i+1))
        frequency.append(num.count(i+1))

    
#터틀그래픽으로 통계 표시 하기 (100 이상 입력 , 폰트 추가, 색 추가, 빈도수에 따른 폰트 크기 추가)
def draw_dice_histogram():    
    t = turtle.Turtle()
    t.goto(-100,0)
    t.speed(0)

    for i in range(6):
        t.fd(20)
        t.pencolor('black')
        t.pu()
        t.rt(90)
        t.fd(20)
        t.write(str(i+1),font=("Arail"))
        t.backward(20)
        t.lt(90)
        t.pd()
        t.pencolor(color[i])
        t.begin_fill()
        t.fillcolor(color[i])
        t.lt(90)
        t.fd(frequency[i]*5)
        t.write(frequency[i],font=("나눔고딕",frequency[i]))
        t.rt(90)
        t.fd(30)
        t.rt(90)
        t.fd(frequency[i]*5)
        t.lt(90)
        t.end_fill()

#동전 던지기

def coin():
    for i in range (freq):
        coin = random.randint(0,1)
        coin1.append(coin)

    for i in range (2):
        print (i,"이(가) 나온 횟수 = ",coin1.count(i))
        coin_frequency.append(coin1.count(i))

# 동전 던지기 그래프 그리기

def draw_coin_histogram():
    t = turtle.Turtle()
    
    for i in range(0,1):
        t.write(coin1.count(i))
        t.fd(coin_frequency[i]*30)
        t.up()
        t.goto(0,-50)
        t.down()
        t.fd(coin_frequency[i+1]*30)

#가위바위보 게임 함수.
def RSP():

    for i in range(freq):

        #플레이어 선택 화면
        player = input("무엇을 낼 것인지 선택하시오! (rock/paper/scissors): ")
        
        while (player != "rock" and player != "paper" and player != "scissors"):
            
            player = input("다시 입력하세요! (rock/paper/scissors): ")
        

        #컴퓨터 가위바위보 선택
        computer = random.randint(1,3)
        if (computer == 1):
            computer = "rock"
        elif (computer == 2):
            computer = "paper"
        elif (computer == 3):
            computer = "scissors"
        else:
            print ("rock sissor paper 다시 입력해주세요~")

        #결과
        if (player == computer):
            
            print ("비겼습니다!!")
            
            result.append(2)
        
        elif (player == "rock"):
            
            if (computer == "paper"):
            
                print ("졌습니다!")
                result.append(1)
        
            if (computer == "scissors"):
            
                print ("이겼습니다!")
                result.append(3)
        elif (player == "paper"):
            
            if (computer == "rock"):
                result.append(1)
                print ("이겼습니다!")
    
            if (computer == "scissors"):
                print ("졌습니다!");
                result.append(3)
    print("이긴 횟수 : ",result.count(3),"회, 비긴 횟수 : ",result.count(2),"회, 진 횟수 : ",result.count(1)," 회 입니다. ")

#가위바위보 게임을 히스토그램으로 나타낸다.
def draw_RSP_histogram():
    t = turtle.Turtle()
    
    for i in range(0,1):
        t.write("이긴 횟수")
        t.fd(result.count(3)*30)
        t.up()
        t.goto(0,-50)
        t.down()
        t.write("비긴 횟수")
        t.fd(result.count(2)*30)
        t.up()
        t.goto(0,-100)
        t.down()
        t.write("진 횟수")
        t.fd(result.count(1)*30)

#메인 함수이다.    
def menu():        
    if select == 1:
        dice_play()
        draw_dice_histogram()
    elif select == 2:
        coin()
        draw_coin_histogram()
    elif select == 3:
        RSP()
        draw_RSP_histogram()
menu()
```
